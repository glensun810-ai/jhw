# 端到端全链路打通方案

**制定时间**: 2026-03-07
**制定人**: 系统首席架构师
**目标**: 一次性彻底打通前端到后端的完整诊断流程，结束反复失败

---

## 一、问题反思：为什么反复失败？

### 1.1 修复方法问题

| 问题 | 表现 | 后果 |
|------|------|------|
| **头痛医头** | 看到 brand 字段缺失就修 brand，看到 tokens=0 就修 tokens | 修好一个，另一个又坏 |
| **缺乏端到端验证** | 只检查后端日志，不验证前端实际输出 | 后端"成功"但前端失败 |
| **假设链条断裂** | 假设 A→B→C 自动流通，实际 B 丢失数据 | 数据在传递中丢失 |
| **缺少可观测性** | 不知道数据在哪一步丢失 | 盲目猜测，反复试错 |

### 1.2 技术债务累积

```
第 1 次修复：brand 字段缺失 → 添加 brand 字段 ✅
第 2 次修复：tokens_used=0 → 添加 tokens_used ✅
第 3 次修复：diagnosis_analysis 缺失 → 添加保存逻辑 ✅
第 4 次修复：??? → ??? 

问题：每次都是"打补丁"，从未验证完整链路！
```

---

## 二、新策略：端到端全链路追踪

### 2.1 核心思想

**不再假设，而是验证每一个环节！**

```
前端请求 → API 接收 → 执行诊断 → 保存结果 → 聚合报告 → 返回前端
   ↓          ↓          ↓          ↓          ↓          ↓
 验证 1     验证 2     验证 3     验证 4     验证 5     验证 6
```

### 2.2 实施方法

#### 方法 1：链路追踪 ID（Trace ID）

每个诊断请求生成唯一追踪 ID，贯穿所有环节：

```python
# 1. 前端生成 trace_id
const traceId = `${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;

// 2. 传递给后端
wx.request({
  url: '/api/perform-brand-test',
  data: {
    trace_id: traceId,  // 传递追踪 ID
    execution_id: executionId,
    ...
  }
});

// 3. 后端所有日志都包含 trace_id
api_logger.info(f"[{trace_id}] 开始诊断...")
api_logger.info(f"[{trace_id}] 保存结果...")
api_logger.info(f"[{trace_id}] 返回报告...")
```

#### 方法 2：关键节点数据快照

在每个关键节点记录数据快照，便于定位丢失位置：

```python
# 节点 1: API 接收数据
log_snapshot("API 接收", {
    "execution_id": execution_id,
    "brand": brand,
    "questions_count": len(questions)
})

# 节点 2: AI 调用完成
log_snapshot("AI 调用完成", {
    "execution_id": execution_id,
    "results_count": len(results),
    "first_result_brand": results[0].get('brand'),
    "first_result_tokens": results[0].get('tokens_used')
})

# 节点 3: 数据库保存后
log_snapshot("数据库保存", {
    "execution_id": execution_id,
    "saved_count": db_count,
    "first_row_brand": db_first.brand
})

# 节点 4: 报告聚合后
log_snapshot("报告聚合", {
    "execution_id": execution_id,
    "report_score": report.get('overallScore'),
    "has_analysis": report.get('brandAnalysis') is not None
})

# 节点 5: 返回前端前
log_snapshot("返回前端", {
    "execution_id": execution_id,
    "response_size": len(json_response),
    "success": True
})
```

#### 方法 3：数据完整性断言

在关键节点添加强断言，失败立即报错而非继续：

```python
def _validate_results_before_save(results: List[Dict]) -> None:
    """保存前强制验证数据完整性"""
    assert len(results) > 0, "结果列表不能为空"
    
    for i, r in enumerate(results):
        assert 'brand' in r and r['brand'], f"结果{i}: brand 字段缺失或为空"
        assert 'question' in r, f"结果{i}: question 字段缺失"
        assert 'model' in r, f"结果{i}: model 字段缺失"
        assert 'response' in r, f"结果{i}: response 字段缺失"
        # 【P1 验证】tokens_used 验证
        assert 'tokens_used' in r, f"结果{i}: tokens_used 字段缺失"
        
    api_logger.info(f"✅ 数据完整性验证通过：{len(results)}条结果")
```

## 三、最小可验证路径（MVP）

### 3.1 核心路径定义

**先确保最简单的场景能工作，再增加复杂度！**

```
┌─────────────────────────────────────────────────────────┐
│  MVP 场景：1 个问题 × 1 个模型 × 1 个品牌                  │
├─────────────────────────────────────────────────────────┤
│  1. 前端发起诊断请求（1 个问题，1 个模型）                │
│  2. 后端调用 1 次 AI                                      │
│  3. 保存 1 条结果到 diagnosis_results                    │
│  4. 返回最简单的报告（只有分数，不要复杂分析）           │
│  5. 前端显示报告                                         │
└─────────────────────────────────────────────────────────┘
```

### 3.2 MVP 验证清单

| 步骤 | 验证点 | 验证方法 | 通过标准 |
|------|--------|----------|----------|
| 1 | 前端请求发送 | 小程序开发者工具 Network 面板 | 看到请求，状态码 200 |
| 2 | 后端 API 接收 | 后端日志 | 看到"收到诊断请求"日志 |
| 3 | AI 调用执行 | 后端日志 | 看到"AI 调用成功"日志 |
| 4 | 结果保存数据库 | 数据库查询 | `SELECT COUNT(*) FROM diagnosis_results WHERE execution_id=?` 返回 1 |
| 5 | 报告生成 | 后端日志 | 看到"报告聚合完成"日志 |
| 6 | 返回前端 | Network 面板 | 响应体包含 `overallScore` |
| 7 | 前端显示 | 小程序界面 | 看到分数显示 |

### 3.3 MVP 调试脚本

创建专用调试脚本，一键验证完整链路：

```python
# scripts/end_to_end_test.py
"""
端到端测试脚本 - 验证完整诊断链路

使用方法:
    cd backend_python
    python3 scripts/end_to_end_test.py
"""

import requests
import time
import sqlite3

BASE_URL = "http://localhost:5000"
DB_PATH = "database.db"

def test_end_to_end():
    """端到端测试"""
    print("=" * 60)
    print("端到端诊断测试")
    print("=" * 60)
    
    # Step 1: 发起诊断请求
    print("\n[Step 1] 发起诊断请求...")
    response = requests.post(f"{BASE_URL}/api/perform-brand-test", json={
        "user_id": "test_user",
        "brand_name": "趣车良品",
        "competitor_brands": ["车尚艺"],
        "selected_models": [{"name": "qwen", "checked": True}],
        "questions": ["深圳新能源汽车改装门店靠谱的推荐？"]
    })
    
    assert response.status_code == 200, f"请求失败：{response.status_code}"
    data = response.json()
    execution_id = data.get('execution_id')
    print(f"✅ 请求成功，execution_id={execution_id}")
    
    # Step 2: 等待诊断完成
    print("\n[Step 2] 等待诊断完成...")
    for i in range(30):  # 最多等待 30 秒
        status_response = requests.get(f"{BASE_URL}/api/diagnosis-status/{execution_id}")
        status = status_response.json()
        
        if status.get('status') == 'completed':
            print(f"✅ 诊断完成，耗时={status.get('elapsed_time', 'N/A')}秒")
            break
        
        print(f"  当前状态：{status.get('status')} ({i+1}/30)")
        time.sleep(1)
    else:
        print("❌ 诊断超时")
        return False
    
    # Step 3: 验证数据库结果
    print("\n[Step 3] 验证数据库结果...")
    conn = sqlite3.connect(DB_PATH)
    cursor = conn.cursor()
    
    cursor.execute("""
        SELECT execution_id, brand, model, tokens_used, created_at 
        FROM diagnosis_results 
        WHERE execution_id = ?
    """, (execution_id,))
    
    results = cursor.fetchall()
    print(f"  数据库结果数：{len(results)}")
    
    if len(results) == 0:
        print("❌ 数据库无结果")
        conn.close()
        return False
    
    # 验证第一条结果
    row = results[0]
    print(f"  第一条结果：brand={row[1]}, model={row[2]}, tokens_used={row[3]}")
    
    if not row[1]:  # brand 为空
        print("❌ brand 字段为空")
        conn.close()
        return False
    
    if row[3] == 0:  # tokens_used 为 0
        print("⚠️  warning: tokens_used=0")
    
    conn.close()
    print("✅ 数据库验证通过")
    
    # Step 4: 获取最终报告
    print("\n[Step 4] 获取最终报告...")
    report_response = requests.get(f"{BASE_URL}/api/diagnosis-report/{execution_id}")
    report = report_response.json()
    
    print(f"  报告分数：{report.get('overallScore', 'N/A')}")
    print(f"  品牌分析：{'✅' if report.get('brandAnalysis') else '❌'}")
    print(f"  竞争分析：{'✅' if report.get('competitiveAnalysis') else '❌'}")
    
    print("\n" + "=" * 60)
    print("✅ 端到端测试通过")
    print("=" * 60)
    return True

if __name__ == "__main__":
    success = test_end_to_end()
    exit(0 if success else 1)
```

## 五、前端调试增强

### 5.1 前端请求追踪

```javascript
// pages/index/index.js - 增强版
Page({
  data: {
    traceId: '',
    diagnosisLog: []  // 本地日志
  },

  async startDiagnosis() {
    // 生成追踪 ID
    const traceId = `${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
    this.setData({ traceId });
    
    this.addLog(`[前端] 开始诊断，traceId=${traceId}`);
    
    try {
      // Step 1: 发起请求
      this.addLog(`[前端] 发起 API 请求...`);
      
      const response = await wx.request({
        url: 'http://localhost:5000/api/perform-brand-test',
        method: 'POST',
        data: {
          trace_id: traceId,  // 传递追踪 ID
          user_id: 'test_user',
          brand_name: this.data.brandName,
          // ...
        },
        success: (res) => {
          this.addLog(`[前端] ✅ 请求成功，status=${res.statusCode}`);
          this.addLog(`[前端] 响应数据：${JSON.stringify(res.data).substr(0, 200)}...`);
        },
        fail: (err) => {
          this.addLog(`[前端] ❌ 请求失败：${JSON.stringify(err)}`);
        }
      });
      
      const executionId = response.data.execution_id;
      this.setData({ executionId });
      
      // Step 2: 轮询状态
      this.pollStatus(executionId, traceId);
      
    } catch (error) {
      this.addLog(`[前端] ❌ 异常：${error.message}`);
      console.error(error);
    }
  },

  async pollStatus(executionId, traceId) {
    this.addLog(`[前端] 开始轮询状态...`);
    
    for (let i = 0; i < 60; i++) {
      try {
        const statusRes = await wx.request({
          url: `http://localhost:5000/api/diagnosis-status/${executionId}`,
          success: (res) => {
            const status = res.data;
            this.addLog(`[前端] 轮询 ${i+1}/60: status=${status.status}, progress=${status.progress}`);
            
            if (status.status === 'completed') {
              this.addLog(`[前端] ✅ 诊断完成！`);
              this.loadReport(executionId);
              return;
            }
          }
        });
      } catch (err) {
        this.addLog(`[前端] 轮询失败：${err.message}`);
      }
      
      await new Promise(resolve => setTimeout(resolve, 2000));
    }
    
    this.addLog(`[前端] ❌ 轮询超时`);
  },

  async loadReport(executionId) {
    this.addLog(`[前端] 加载报告...`);
    
    const reportRes = await wx.request({
      url: `http://localhost:5000/api/diagnosis-report/${executionId}`,
      success: (res) => {
        this.addLog(`[前端] ✅ 报告加载成功`);
        this.addLog(`[前端] 报告内容：${JSON.stringify(res.data).substr(0, 300)}...`);
        
        // 设置报告数据
        this.setData({
          reportData: res.data,
          showReport: true
        });
      },
      fail: (err) => {
        this.addLog(`[前端] ❌ 报告加载失败：${err.message}`);
      }
    });
  },

  addLog(message) {
    const timestamp = new Date().toLocaleTimeString();
    const log = `[${timestamp}] ${message}`;
    this.data.diagnosisLog.push(log);
    this.setData({ diagnosisLog: this.data.diagnosisLog });
    console.log(log);
  }
})
```
