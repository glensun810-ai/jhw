```python
@contextmanager
def get_connection(self):
    conn = None
    try:
        conn = sqlite3.connect(self.db_path, timeout=30)
        logger.debug(f"创建数据库连接：{id(conn)}")
        yield conn
    except Exception as e:
        logger.error(f"数据库操作失败：{e}")
        if conn:
            conn.rollback()
        raise
    finally:
        if conn:
            conn.close()
            logger.debug(f"关闭数据库连接：{id(conn)}")
```

```python
def update_report_status(self, execution_id, status, stage):
```

```python
# utils/data_transformer.py

def to_camel_case(snake_str):
    """将 snake_case 转换为 camelCase"""
    components = snake_str.split('_')
    return components[0] + ''.join(x.title() for x in components[1:])

def convert_dict_to_camel(data):
    """递归转换字典键名"""
    if isinstance(data, dict):
        return {to_camel_case(k): convert_dict_to_camel(v) for k, v in data.items()}
    elif isinstance(data, list):
        return [convert_dict_to_camel(item) for item in data]
    else:
        return data

# 在 API 层统一使用
@diagnosis_bp.route('/report/<execution_id>')
def get_full_report(execution_id):
    report = service.get_full_report(execution_id)
    return jsonify(convert_dict_to_camel(report))
```
