```python
# 在 app.py 启动检查中添加
def validate_api_keys():
    """验证 API Key 是否有效"""
    providers = ['qwen', 'doubao', 'deepseek']
    for provider in providers:
        try:
            # 发送测试请求
            test_response = provider.health_check()
            if not test_response.success:
                api_logger.error(f"{provider} API Key 验证失败")
        except Exception as e:
            api_logger.error(f"{provider} API Key 验证异常：{e}")
```
