**关键信息**：

- 重试次数：17/18 次
- 错误类型：`request:fail timeout`

### 1.2 根本原因分析

**问题 1: `/test/status` 端点性能瓶颈**

当前实现（`diagnosis_views.py:2740`）：
```python
@wechat_bp.route('/test/status/<task_id>', methods=['GET'])
def get_task_status_api(task_id):
    # 调用 get_full_report，需要查询 3 个表
    service = get_report_service()
    report = service.get_full_report(task_id)  # ❌ 重查询
```

`get_full_report` 方法（`diagnosis_report_service.py:180`）：
```python
def get_full_report(self, execution_id: str):
    # 1. 查询报告主表
    report = self.report_repo.get_by_execution_id(execution_id)
    
    # 2. 查询结果明细表

    sentiment_distribution = self._calculate_sentiment_distribution(results)
```

**性能问题**：
- 每次轮询都需要查询 3 个表
- 需要计算多个统计指标
- 前端每 800ms 轮询一次，造成数据库压力

**问题 2: 数据库连接池竞争**

```python
# database_connection_pool.py

    # 如果连接池已满，需要等待
    # 高并发时可能导致超时
```

**问题 3: 状态不一致导致的重复查询**

从之前修复的代码看，`/api/diagnosis/status/{execution_id}` 端点需要：
```python

# 2. 获取数据库状态

# 3. 获取数据库结果

```

这进一步增加了数据库查询负担。

### 方案 A: 优化 `/test/status` 端点（推荐）

**核心思路**：简化查询，只返回必要字段

**修改文件**: `wechat_backend/views/diagnosis_views.py`

**修改内容**：

```python
@wechat_bp.route('/test/status/<task_id>', methods=['GET'])
@rate_limit(limit=30, window=60, per='endpoint')  # 【优化】提高限流
@monitored_endpoint('/test/status', require_auth=False, validate_inputs=False)
def get_task_status_api(task_id):
    """
    轮询任务进度与分阶段状态（P0 性能优化版 - 2026-03-04）

    【P0 关键修复】
    1. 只查询报告主表，不查询明细和结果
    2. 使用轻量级查询，减少数据库负载
    3. 添加缓存层，避免重复查询
    4. 增量轮询优化，无变化时返回空响应
    """
    if not task_id:
        return jsonify({'error': 'Task ID is required'}), 400

    # 获取增量轮询参数
    since = request.args.get('since')
    fields = request.args.get('fields', 'basic')  # basic | full

    try:
        from wechat_backend.diagnosis_report_repository import DiagnosisReportRepository
        
        report_repo = DiagnosisReportRepository()
        
        # 【P0 优化】只查询报告主表，不查询明细

        if not report:
            # 降级：从 execution_store 读取
            if task_id in execution_store:
                report = execution_store[task_id]
            else:
                return jsonify({
                    'task_id': task_id,
                    'status': 'not_found',
                    'progress': 0,
                    'error': 'Task not found'
                }), 404

        # 提取基本字段
        stage = report.get('stage', 'init')
        status = report.get('status', 'processing')
        progress = report.get('progress', 0)
        is_completed = bool(report.get('is_completed', False))
        updated_at = report.get('updated_at', '')

        # 【P0 优化】增量轮询：无变化时返回空响应
        if since and updated_at <= since:
            return jsonify({
                'task_id': task_id,
                'has_updates': False,
                'last_updated': updated_at,
                'status': status,
                'progress': progress
            }), 200

        # 【P0 修复】状态推导（防止一直是 init）
        if stage == 'init' and progress > 0:
            stage = 'ai_fetching'
        if progress >= 100 and stage not in ['completed', 'failed']:
            stage = 'completed'
            status = 'completed'
            is_completed = True

        # 构建轻量级响应
        response_data = {
            'task_id': task_id,
            'status': status,
            'stage': stage,
            'progress': progress,
            'is_completed': is_completed,
            'should_stop_polling': status in ['completed', 'failed'],
            'updated_at': updated_at,
            'has_updates': True,
            'source': 'database'
        }

        # 【优化】只在完成时返回结果数据
        if is_completed and fields == 'full':
            from wechat_backend.diagnosis_report_repository import DiagnosisResultRepository
            result_repo = DiagnosisResultRepository()
            results = result_repo.get_by_execution_id(task_id)
            response_data['results'] = results

        # 【P0 修复】添加防缓存头
        response = jsonify(response_data)
        response.headers['Cache-Control'] = 'no-cache, no-store, must-revalidate'
        response.headers['Pragma'] = 'no-cache'
        response.headers['Expires'] = '0'
        
        api_logger.info(
            f"[TaskStatus-Lite] 返回数据：{task_id}, "
            f"status={status}, progress={progress}, source=database"
        )
        
        return response, 200

    except Exception as e:
        api_logger.error(f'[TaskStatus] 查询失败：{task_id}, 错误：{e}')
        
        # 降级：从 execution_store 读取
        if task_id in execution_store:
            task_status = execution_store[task_id]
            return jsonify({
                'task_id': task_id,
                'status': task_status.get('status', 'unknown'),
                'stage': task_status.get('stage', 'init'),
                'progress': task_status.get('progress', 0),
                'is_completed': task_status.get('is_completed', False),
                'should_stop_polling': task_status.get('status') in ['completed', 'failed'],
                'source': 'cache'
            }), 200
        
        return jsonify({
            'task_id': task_id,
            'status': 'error',
            'error': str(e)
        }), 500
```

### 方案 B: 添加 Redis 缓存层

**核心思路**：缓存任务状态，减少数据库查询

**修改文件**: `wechat_backend/cache/task_status_cache.py` (新建)

```python
"""
任务状态缓存 - P0 性能优化

核心思路：
1. 使用 Redis 缓存任务状态
2. 诊断执行时同步更新缓存
3. 轮询端点直接从缓存读取
"""

import json
from typing import Dict, Any, Optional
from datetime import datetime, timedelta
from wechat_backend.logging_config import api_logger

try:
    from wechat_backend.cache.redis_cache import get_redis_client
    REDIS_AVAILABLE = True
except ImportError:
    REDIS_AVAILABLE = False


class TaskStatusCache:
    """任务状态缓存"""

    def __init__(self):
        self.redis_client = get_redis_client() if REDIS_AVAILABLE else None
        self.prefix = 'task_status:'
        self.ttl_seconds = 600  # 10 分钟 TTL

    def set_status(self, task_id: str, status_data: Dict[str, Any]):
        """
        设置任务状态

        Args:
            task_id: 任务 ID
            status_data: 状态数据
        """
        if not self.redis_client:
            return

        try:
            key = f"{self.prefix}{task_id}"
            value = json.dumps({
                **status_data,
                '_cached_at': datetime.now().isoformat()
            })
            self.redis_client.setex(key, self.ttl_seconds, value)
            api_logger.debug(f"[TaskStatusCache] 缓存已设置：{task_id}")
        except Exception as e:
            api_logger.error(f"[TaskStatusCache] 设置缓存失败：{task_id}, 错误：{e}")

    def get_status(self, task_id: str) -> Optional[Dict[str, Any]]:
        """
        获取任务状态

        Args:
            task_id: 任务 ID

        Returns:
            状态数据，不存在返回 None
        """
        if not self.redis_client:
            return None

        try:
            key = f"{self.prefix}{task_id}"
            value = self.redis_client.get(key)
            if value:
                data = json.loads(value)
                api_logger.debug(f"[TaskStatusCache] 缓存命中：{task_id}")
                return data
            api_logger.debug(f"[TaskStatusCache] 缓存未命中：{task_id}")
            return None
        except Exception as e:
            api_logger.error(f"[TaskStatusCache] 获取缓存失败：{task_id}, 错误：{e}")
            return None

    def delete_status(self, task_id: str):
        """删除任务状态缓存"""
        if not self.redis_client:
            return

        try:
            key = f"{self.prefix}{task_id}"
            self.redis_client.delete(key)
            api_logger.debug(f"[TaskStatusCache] 缓存已删除：{task_id}")
        except Exception as e:
            api_logger.error(f"[TaskStatusCache] 删除缓存失败：{task_id}, 错误：{e}")


# 全局单例
_task_status_cache: Optional[TaskStatusCache] = None


def get_task_status_cache() -> TaskStatusCache:
    """获取任务状态缓存实例"""
    global _task_status_cache
    if _task_status_cache is None:
        _task_status_cache = TaskStatusCache()
    return _task_status_cache
```
