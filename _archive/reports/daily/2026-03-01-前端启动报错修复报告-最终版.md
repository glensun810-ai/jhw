TypeError: (that.data.domesticAiModels || []).map is not a function

---

## 根本原因分析

### 问题 1: data 对象未初始化数组
`domesticAiModels` 和 `overseasAiModels` 在 `data` 对象中**没有定义**，导致：
- 初始访问时返回 `undefined`
- `(undefined || [])` 返回 `[]`（空数组）
- 但在某些情况下，数据可能被设为其他类型（如对象）

### 问题 2: 异步时序问题
```javascript
onLoad: function() {
  this.initDomesticAiModels();  // ← 异步 setData
  this.loadPlatformConfigStatus();  // ← 立即执行，可能先于上面完成
}
```

### 问题 3: wx.request 回调中的数据变化
在异步回调中，`that.data` 可能已经被其他操作修改。

---

## 完整修复方案

### 修复 1: 在 data 中初始化数组

```javascript
data: {
  // ... 其他数据

  // 【P0 新增】AI 平台列表 - 初始化为空数组
  domesticAiModels: [],
  overseasAiModels: []
}
```

**作用**: 确保变量始终存在，即使是空数组。

### 修复 2: 添加严格的类型检查

```javascript
loadPlatformConfigStatus: function() {
  const that = this;

  // 防御性检查：确保 domesticAiModels 是数组
  if (!Array.isArray(that.data.domesticAiModels)) {
    console.warn('[平台配置] domesticAiModels 不是数组 (类型：' + 
      typeof that.data.domesticAiModels + ')，等待初始化完成');
    setTimeout(() => {
      this.loadPlatformConfigStatus();
    }, 100);
    return;
  }

  // 防御性检查：确保 overseasAiModels 是数组
  if (!Array.isArray(that.data.overseasAiModels)) {
    console.warn('[平台配置] overseasAiModels 不是数组 (类型：' + 
      typeof that.data.overseasAiModels + ')，等待初始化完成');
    setTimeout(() => {
      this.loadPlatformConfigStatus();
    }, 100);
    return;
  }

  // ... 继续执行
}
```

**作用**: 
- 检查是否为数组类型
- 记录实际类型便于调试
- 延迟重试直到数据准备好

### 修复 3: 使用 Array.from 确保数组

```javascript
// 在请求前保存数组引用，避免回调中数据变化
// 使用 Array.from 确保是真正的数组
const currentDomesticModels = Array.from(that.data.domesticAiModels || []);
const currentOverseasModels = Array.from(that.data.overseasAiModels || []);
```

**作用**: 
- `Array.from()` 将类数组对象转换为真数组
- 即使传入的是对象或其他类型，也会返回空数组
- 比 `(x || [])` 更安全

### 修复 4: 添加调试日志

**作用**: 便于追踪数据状态。

## 完整修复代码

```javascript
/**
 * 【新增】加载平台配置状态，标记未配置的平台
 */
loadPlatformConfigStatus: function() {
  const that = this;

  // 防御性检查：确保 domesticAiModels 和 overseasAiModels 是数组
  if (!Array.isArray(that.data.domesticAiModels)) {
    console.warn('[平台配置] domesticAiModels 不是数组 (类型：' + typeof that.data.domesticAiModels + ')，等待初始化完成');
    // 延迟重试
    setTimeout(() => {
      this.loadPlatformConfigStatus();
    }, 100);
    return;
  }

  if (!Array.isArray(that.data.overseasAiModels)) {
    console.warn('[平台配置] overseasAiModels 不是数组 (类型：' + typeof that.data.overseasAiModels + ')，等待初始化完成');
    // 延迟重试
    setTimeout(() => {
      this.loadPlatformConfigStatus();
    }, 100);
    return;
  }

  // 在请求前保存数组引用，避免回调中数据变化
  // 使用 Array.from 确保是真正的数组
  const currentDomesticModels = Array.from(that.data.domesticAiModels || []);
  const currentOverseasModels = Array.from(that.data.overseasAiModels || []);

  console.log('[平台配置] 开始加载平台状态，国内平台数：' + currentDomesticModels.length + ', 海外平台数：' + currentOverseasModels.length);

  wx.request({
    url: getApp().globalData.serverUrl + '/api/platform-status',
    method: 'GET',
    success: function(res) {
      if (res.statusCode === 200 && res.data.status === 'success') {
        const platforms = res.data.platforms;

        // 更新国内平台配置状态（使用保存的数组引用）
        const domesticAiModels = currentDomesticModels.map(model => {
          const platformStatus = platforms[model.id];
          return {
            ...model,
            isConfigured: platformStatus ? platformStatus.isConfigured : false,
            disabled: platformStatus ? !platformStatus.isConfigured : true
          };
        });

        // 更新海外平台配置状态（使用保存的数组引用）
        const overseasAiModels = currentOverseasModels.map(model => {
          const platformStatus = platforms[model.id];
          return {
            ...model,
            isConfigured: platformStatus ? platformStatus.isConfigured : false,
            disabled: platformStatus ? !platformStatus.isConfigured : true
          };
        });

        that.setData({
          domesticAiModels,
          overseasAiModels
        });

        console.log('[平台配置] 已更新平台配置状态');
      } else {
        console.warn('[平台配置] 响应状态异常:', res.statusCode);
      }
    },
    fail: function(err) {
      console.warn('[平台配置] 获取平台状态失败:', err);
    }
  });
}
```
