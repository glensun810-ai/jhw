```python
def enrich_dashboard_with_brand_analysis(dashboard_data: dict, execution_id: str) -> dict:
    """P1 级修复：为 Dashboard 数据添加品牌分析相关字段"""
    
    # 从 execution_store 获取后台分析结果
    execution_store = get_execution_store()
    exec_data = execution_store.get(execution_id, {})
    final_report = exec_data.get('final_report', {})
    
    # 提取品牌分析数据
    if final_report:
        dashboard_data['brandAnalysis'] = final_report.get('brandAnalysis')
        dashboard_data['userBrandAnalysis'] = final_report.get('userBrandAnalysis')
        dashboard_data['competitorAnalysis'] = final_report.get('competitorAnalysis', [])
        dashboard_data['comparison'] = final_report.get('comparison')
        dashboard_data['top3Brands'] = final_report.get('top3Brands', [])
    
    return dashboard_data
```
