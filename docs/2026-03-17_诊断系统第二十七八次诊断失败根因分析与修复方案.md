**文件**: `backend_python/wechat_backend/services/brand_analysis_service.py`

```python
def extract_keywords(self, content: str, top_k: int = 10) -> List[str]:
    """
    从 AI 响应内容中提取关键词
    
    参数:
        content: AI 响应内容
        top_k: 返回的关键词数量
    
    返回:
        关键词列表
    """
    if not content:
        return []
    
    # 方案 1: 使用简单的文本分析
    # 移除常见停用词
    stop_words = {'的', '了', '是', '在', '我', '有', '和', '就', '不', '人'}
    
    # 提取名词（简单实现：提取 2-4 字短语）
    import re
    phrases = re.findall(r'[\u4e00-\u9fa5]{2,4}', content)
    
    # 统计频率
    from collections import Counter
    word_freq = Counter(phrases)
    
    # 过滤停用词
    filtered_words = [
        word for word, freq in word_freq.most_common(top_k * 2)
        if word not in stop_words
    ]
    
    return filtered_words[:top_k]
```

```python
    # 统计每个品牌的结果数量
    brand_counts = {}
    for result in results:
        brand = result.get('extracted_brand') or result.get('brand')
        if brand:
            brand_counts[brand] = brand_counts.get(brand, 0) + 1
    
    if not brand_counts:
        db_logger.warning("未提取到任何品牌")
        return {
            'data': {},
            'totalCount': 0,
            'successRate': 0.0,
            'warnings': ['未提取到品牌信息']
        }
    
    total = sum(brand_counts.values())
    
    return {
        'data': brand_counts,
        'totalCount': total,
        'successRate': 1.0,
        'brands': list(brand_counts.keys())
    }
```
