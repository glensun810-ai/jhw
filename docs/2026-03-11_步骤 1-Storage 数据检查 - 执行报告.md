**检查脚本**（复制粘贴）:
```javascript
(function() {
  console.log('========================================');
  console.log('📊 Storage 数据检查 - 步骤 1');
  console.log('========================================\n');
  
  const info = wx.getStorageInfoSync();
  console.log('【1】Storage 使用量:', info.currentSize, 'KB /', info.limitSize, 'KB');
  console.log('【2】所有 keys 数量:', info.keys.length);
  
  const diagnosisKeys = info.keys.filter(k => k.startsWith('diagnosis_result_'));
  console.log('【3】诊断 keys:', diagnosisKeys);
  console.log('【4】诊断 keys 数量:', diagnosisKeys.length);
  
  if (diagnosisKeys.length > 0) {
    const data = wx.getStorageSync(diagnosisKeys[0]);
    console.log('\n========================================');
    console.log('【5】最新诊断数据详情');
    console.log('========================================');
    console.log('Key:', diagnosisKeys[0]);
    console.log('ExecutionId:', data.executionId);
    console.log('BrandName:', data.brandName);
    console.log('Timestamp:', new Date(data.timestamp).toLocaleString());
    
    console.log('\n【6】核心数据检查');
    console.log('brandDistribution:', data.data?.brandDistribution ? '✅ 存在' : '❌ 不存在');
    console.log('brandScores:', data.data?.brandScores ? '✅ 存在' : '❌ 不存在');
    console.log('detailedResults:', data.data?.detailedResults?.length || 0, '条');
    console.log('overallScore:', data.data?.overallScore || '未设置');
    
    if (data.data?.brandDistribution) {
      console.log('\n【7】品牌分布详情');
      console.log('total_count:', data.data.brandDistribution.total_count);
      console.log('data keys:', Object.keys(data.data.brandDistribution.data || {}));
    }
    
    console.log('\n========================================');
    console.log('【8】下一步建议');
    console.log('========================================');
    if (!data.data?.brandDistribution && !data.data?.brandScores) {
      console.log('➡️  跳转到步骤 4: 检查后端数据生成');
    } else if (!data.data?.detailedResults || data.data.detailedResults.length === 0) {
      console.log('➡️  跳转到步骤 5: 检查诊断执行流程');
    } else {
      console.log('➡️  继续步骤 2: 检查页面加载逻辑');
    }
  } else {
    console.log('\n❌ 未找到诊断数据！');
    console.log('可能原因:');
    console.log('  1. 未执行诊断');
    console.log('  2. 诊断完成后未保存到 storage');
    console.log('  3. storage key 命名不一致');
    console.log('\n➡️  检查 pages/index/index.js 的 handleDiagnosisComplete 函数');
  }
  
  console.log('\n========================================');
})();
```
