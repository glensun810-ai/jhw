```javascript
async getFullReport(executionId, options = {}) {
  const retryCount = options.retryCount || 0;
  
  // ✅ 【死循环修复 - 2026-03-14】限制最大重试次数为 3 次
  const maxRetryCount = 3;
  
  try {
    // ... API 调用
    timeout: 15000  // ✅ 【优化】15 秒超时
    
  } catch (error) {
    // ✅ 【死循环修复 - 2026-03-14】重试逻辑优化
    const errorType = this._classifyError(error);
    const canRetry = retryCount < maxRetryCount &&
                     errorType !== ErrorTypes.DATA_NOT_FOUND &&
                     errorType !== ErrorTypes.DATA_INVALID;
    
    if (canRetry) {
      const retryDelay = 1000 * (retryCount + 1); // ✅ 指数退避：1s, 2s, 3s
      console.log(`[ReportService] ${retryDelay}ms 后重试：${retryCount + 1}/${maxRetryCount}`);
      await new Promise(resolve => setTimeout(resolve, retryDelay));
      return this.getFullReport(executionId, { retryCount: retryCount + 1 });
    }
    
    // ✅ 达到最大重试次数
    console.error('[ReportService] 最终失败:', errorMessage);
    return this._createEmptyReportWithSuggestion(errorMessage, errorType);
  }
}
```

**验证结果**: ✅ 修复正确应用

```javascript
        if (res.confirm) {
          // ✅ 【修复】使用页面重新加载而非递归调用
```
