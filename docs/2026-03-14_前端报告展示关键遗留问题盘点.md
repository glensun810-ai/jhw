```javascript
_normalizeKeywords(keywords) {
  if (!keywords) return [];
  
  if (Array.isArray(keywords)) {
    if (typeof keywords[0] === 'string') {
      // 字符串数组转为对象数组
      return keywords.map(word => ({ word, weight: 1.0 }));
```

```javascript
// 优化重试逻辑
async getFullReport(executionId, options = {}) {
  const retryCount = options.retryCount || 0;
  const maxRetryCount = 3;
  const retryDelay = 1000 * (retryCount + 1); // 指数退避
  
  try {
    // ... API 调用
  } catch (error) {
    if (retryCount < maxRetryCount && this._canRetry(error)) {
      await new Promise(resolve => setTimeout(resolve, retryDelay));
      return this.getFullReport(executionId, { retryCount: retryCount + 1 });
    }
    throw error;
  }
}
```
