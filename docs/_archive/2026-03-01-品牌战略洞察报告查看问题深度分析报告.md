```javascript
// 新增：带重试的拉取方法
fetchResultsFromServerWithRetry: function(executionId, brandName, retryCount = 3) {
    let attempts = 0;
    
    const tryFetch = () => {
        attempts++;
        fetchResultsService(
            executionId,
            brandName,
            (data) => {
                // 成功
                this.initializePageWithData(data.results, ...);
            },
            (error) => {
                // 失败，尝试重试
                if (attempts < retryCount) {
                    console.log(`[重试] 第${attempts}次尝试`);
                    setTimeout(tryFetch, 1000 * attempts);  // 指数退避
                } else {
                    // 所有重试失败
                    this.showErrorMessage('数据加载失败，请检查网络连接或重试');
                }
            }
        );
    };
    
    tryFetch();
}
```

```javascript
        success: (res) => {
            if (res.statusCode === 403) {
                console.warn('⚠️ Token 已过期 (403)');
                // 降级：尝试使用匿名访问
                wx.removeStorageSync('access_token');
                return fetchResultsFromServer(executionId, brandName, onSuccess, onError);
            }
            
            if (res.statusCode === 200 && res.data) {
                const resultsToUse = res.data.detailed_results || res.data.results || [];
                
                if (resultsToUse.length === 0) {
                    // 降级：结果为空，显示空状态
                    if (onError) onError({ 
                        type: 'empty', 
                        message: '暂无诊断结果',
                        suggestion: '请重新进行诊断'
                    });
                    return;
                }
                
                // 正常处理
                const competitiveAnalysisToUse = res.data.competitive_analysis || {};
                // ... 保存到 Storage ...
                
                if (onSuccess) onSuccess({
                    results: resultsToUse,
                    competitiveAnalysis: competitiveAnalysisToUse,
                    targetBrand: brandName
                });
            } else if (res.statusCode === 404) {
                // 降级：404，显示友好提示
                if (onError) onError({ 
                    type: 'not_found', 
                    message: '未找到诊断结果',
                    suggestion: '报告可能已被删除或执行 ID 错误'
                });
            } else {
                // 降级：其他错误
                if (onError) onError({ 
                    type: 'error', 
                    message: '数据加载失败',
                    code: res.statusCode
                });
            }
        },
        fail: (err) => {
            // 降级：网络失败，建议使用缓存
            const cachedData = loadResultFromStorage(executionId, brandName);
            if (cachedData.useStorageData) {
                console.log('⚠️ 网络失败，使用缓存数据');
                if (onSuccess) onSuccess(cachedData);
            } else {
                if (onError) onError({ 
                    type: 'network', 
                    message: '网络连接失败',
                    suggestion: '请检查网络连接后重试'
                });
            }
        }
    });
};
```

```python
class ReportValidationService:
    def validate_report(self, report: Dict) -> Dict:
        errors = []
        warnings = []
        
        # 1. 检查必需字段
        required_fields = ['report', 'results', 'analysis', 'meta']
        for field in required_fields:
            if field not in report:
                errors.append(f'缺少必需字段：{field}')
        
        if errors:
            return {'is_valid': False, 'errors': errors, 'warnings': []}
        
        # 2. 验证报告主数据
        report_data = report['report']
        if not report_data.get('execution_id'):
            errors.append('execution_id 缺失')
        if not report_data.get('brand_name'):
            errors.append('brand_name 缺失')
        
        # 3. 验证结果数据
        results = report['results']
        if not isinstance(results, list):
            errors.append('results 必须是数组')
        elif len(results) == 0:
            warnings.append('结果数组为空')
        
        # 4. 验证校验和
        checksum = report_data.get('checksum')
        if checksum:
            calculated_checksum = calculate_checksum(report)
            if checksum != calculated_checksum:
                errors.append('校验和不匹配，数据可能已损坏')
        
        # 5. 验证分析数据
        analysis = report['analysis']
        required_analysis = ['competitive_analysis', 'brand_scores']
        for analysis_type in required_analysis:
            if analysis_type not in analysis:
                warnings.append(f'缺少分析数据：{analysis_type}')
        
        return {
            'is_valid': len(errors) == 0,
            'errors': errors,
            'warnings': warnings
        }
```

```python
# 新增：diagnosis_report_transformer.py
class ReportTransformer:
    """报告数据转换器 - 将后端格式转换为前端期望格式"""
    
    @staticmethod
    def transform_to_frontend_format(report: Dict) -> Dict:
        """转换为前端格式"""
        return {
            'brandDistribution': ReportTransformer._calculate_brand_distribution(
                report.get('results', [])
            ),
            'sentimentDistribution': ReportTransformer._calculate_sentiment_distribution(
                report.get('results', [])
            ),
            'keywords': ReportTransformer._extract_keywords(
                report.get('results', [])
            ),
            'competitorAnalysis': report.get('analysis', {}).get('competitive_analysis', {}),
            'brandScores': report.get('analysis', {}).get('brand_scores', {}),
            'semanticDrift': report.get('analysis', {}).get('semantic_drift', {}),
            'sourcePurity': report.get('analysis', {}).get('source_purity', {}),
            'recommendations': report.get('analysis', {}).get('recommendations', {}),
            'meta': report.get('meta', {})
        }
    
    @staticmethod
    def _calculate_brand_distribution(results: List) -> Dict:
        """计算品牌分布"""
        distribution = {}
        for result in results:
            brand = result.get('brand', 'Unknown')
            distribution[brand] = distribution.get(brand, 0) + 1
        
        return {
            'data': distribution,
            'total_count': len(results)
        }
    
    @staticmethod
    def _calculate_sentiment_distribution(results: List) -> Dict:
        """计算情感分布"""
        sentiment_counts = {'positive': 0, 'neutral': 0, 'negative': 0}
        
        for result in results:
            geo_data = result.get('geo_data', {})
            sentiment = geo_data.get('sentiment', 0)
            
            if sentiment > 0.3:
                sentiment_counts['positive'] += 1
            elif sentiment < -0.3:
                sentiment_counts['negative'] += 1
            else:
                sentiment_counts['neutral'] += 1
        
        return {
            'data': sentiment_counts,
            'total_count': len(results)
        }
    
    @staticmethod
    def _extract_keywords(results: List) -> List:
        """提取关键词"""
        keywords = []
        seen = set()
        
        for result in results:
            geo_data = result.get('geo_data', {})
            extracted_keywords = geo_data.get('keywords', [])
            
            for kw in extracted_keywords:
                word = kw.get('word', '')
                if word and word not in seen:
                    keywords.append(kw)
                    seen.add(word)
        
        return keywords
```
