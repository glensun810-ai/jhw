```javascript
async startDiagnosis(params) {
  const serverUrl = app.globalData.serverUrl;
  
  // 2.1 HTTP 请求启动诊断
  const res = await wx.request({
    url: `${serverUrl}/api/perform-brand-test`,
    method: 'POST',
    data: {
      brand_list: params.brandList,
      selected_models: params.selectedModels,
      custom_questions: params.customQuestions,
      user_id: params.userId,
      user_level: params.userLevel
    }
  });

  // 2.2 处理响应
  const { execution_id, report_id } = res.data;
  
  // 2.3 保存到本地存储（用于恢复）
  wx.setStorageSync('pendingDiagnosis', {
    executionId: execution_id,
    reportId: report_id,
    startTime: Date.now()
  });

  return { executionId: execution_id, reportId: report_id };
}
```

```python
@staticmethod
async def start_diagnosis(brand_list, selected_models, custom_questions, user_id, user_level):
    """
    启动品牌诊断
    """
    # 4.1 生成执行 ID
    execution_id = str(uuid.uuid4())

    # 4.2 验证输入
    if not brand_list:
        return {'success': False, 'error': '品牌列表不能为空'}

    # 4.3 提取主品牌和竞品
    main_brand = brand_list[0]
    competitor_brands = brand_list[1:] if len(brand_list) > 1 else []

    # 4.4 执行 NxM 测试
    result = execute_nxm_test(
        execution_id=execution_id,
        main_brand=main_brand,
        competitor_brands=competitor_brands,
        selected_models=selected_models,
        raw_questions=custom_questions,
        user_id=user_id,
        user_level=user_level,
        execution_store={}
    )

    # 4.5 返回执行信息
    return {
        'success': True,
        'execution_id': execution_id,
        'formula': result.get('formula', ''),
        'total_tasks': result.get('total_tasks', 0)
    }
```

## 3. 后端执行引擎流程

### 3.1 NxM 执行引擎入口

**文件**: `backend_python/wechat_backend/nxm_execution_engine.py`

```python
def execute_nxm_test(execution_id, main_brand, competitor_brands, selected_models, 
                     raw_questions, user_id, user_level, execution_store):
    """
    NxM 并发执行引擎
    
    参数:
        execution_id: 执行 ID
        main_brand: 主品牌
        competitor_brands: 竞品列表
        selected_models: AI 模型列表
        raw_questions: 问题列表
        user_id: 用户 ID
        user_level: 用户等级
        execution_store: 执行状态存储
    
    返回:
        {
            'execution_id': str,
            'status': str,
            'progress': int,
            'results': list,
            'formula': str,
            'total_tasks': int
        }
    """
    start_time = time.time()
    
    # 1. 初始化执行状态
    execution_store[execution_id] = {
        'status': 'initializing',
        'progress': 0,
        'stage': 'init',
        'completed': 0,
        'total': len(main_brands) * len(selected_models) * len(raw_questions),
        'results': [],
        'start_time': datetime.now().isoformat()
    }
    
    # 2. 创建诊断报告记录（数据库）
    report_id = create_diagnosis_report(execution_id, user_id, {
        'brand_name': main_brand,
        'competitor_brands': competitor_brands,
        'selected_models': selected_models,
        'custom_questions': raw_questions
    })
    
    # 3. 构建任务矩阵 (NxM)
    tasks = []
    for brand in [main_brand] + competitor_brands:
        for model in selected_models:
            for question in raw_questions:
                tasks.append({
                    'brand': brand,
                    'model': model,
                    'question': question
                })
    
    # 4. 并发执行任务
    for task in tasks:
        try:
            # 4.1 调用 AI 模型
            response = call_ai_model(
                model=task['model'],
                question=task['question'],
                brand=task['brand']
            )
            
            # 4.2 解析 GEO 数据
            geo_data, error = parse_geo_with_validation(
                response_text=response,
                execution_id=execution_id,
                q_idx=tasks.index(task),
                model_name=task['model']['name'],
                brand_name=task['brand']
            )
            
            # 4.3 计算质量评分
            quality = calculate_result_quality(geo_data)
            
            # 4.4 保存结果
            result = {
                'brand': task['brand'],
                'question': task['question'],
                'model': task['model']['name'],
                'response': response,
                'geo_data': geo_data,
                'quality_score': quality['quality_score'],
                'quality_level': quality['quality_level']
            }
            
            execution_store[execution_id]['results'].append(result)
            execution_store[execution_id]['completed'] += 1
            
            # 4.5 更新进度
            execution_store[execution_id]['progress'] = int(
                (execution_store[execution_id]['completed'] / 
                 execution_store[execution_id]['total']) * 100
            )
            
            # 4.6 实时持久化（WAL 预写日志）
            write_wal(execution_id, 
                     execution_store[execution_id]['results'],
                     execution_store[execution_id]['completed'],
                     execution_store[execution_id]['total'])
            
        except Exception as e:
            api_logger.error(f"任务执行失败：{e}")
            # 容错处理：记录错误但继续执行
    
    # 5. 完成执行
    execution_store[execution_id]['status'] = 'completed'
    execution_store[execution_id]['stage'] = 'completed'
    
    return {
        'execution_id': execution_id,
        'status': 'completed',
        'progress': 100,
        'results': execution_store[execution_id]['results'],
        'formula': 'NxM',
        'total_tasks': len(tasks)
    }
```

```python
class DeepSeekAdapter:
    """DeepSeek AI 适配器"""

        )

        return content
```

### 3.3 容错执行器

**文件**: `backend_python/wechat_backend/fault_tolerant_executor.py`

```python
class FaultTolerantExecutor:
    """容错执行器"""
    
    def __init__(self, timeout_seconds=10):
        self.timeout_seconds = timeout_seconds
    
    async def execute_with_fallback(self, task_func, task_name, source, *args, **kwargs):
        """
        带降级的执行
        
        参数:
            task_func: 任务函数
            task_name: 任务名称
            source: 数据源
            *args, **kwargs: 任务参数
        
        返回:
            FaultTolerantResult
        """
        try:
            # 1. 执行任务（带超时）
            result = await asyncio.wait_for(
                task_func(*args, **kwargs),
                timeout=self.timeout_seconds
            )
            
            return FaultTolerantResult.success(
                data=result,
                source=source,
                execution_time=self.timeout_seconds
            )
            
        except asyncio.TimeoutError:
            api_logger.error(f"任务超时：{task_name}")
            return FaultTolerantResult.failed(
                error_message=f"{task_name} 执行超时",
                error_type=ErrorType.TIMEOUT,
                source=source
            )
            
        except Exception as e:
            api_logger.error(f"任务失败：{task_name}, {e}")
            return FaultTolerantResult.failed(
                error_message=str(e),
                error_type=ErrorType.UNKNOWN,
                source=source
            )
```

```python
def parse_geo_with_validation(response_text, execution_id, q_idx, model_name, brand_name):
    """
    解析 GEO 数据并验证
    
    返回:
        geo_data: {
            'brand_mentioned': bool,
            'rank': int,
            'sentiment': float,
            'cited_sources': list,
            'interception': str
        }
        error: 错误信息
    """
    try:
        # 1. 解析 JSON
        geo_data = json.loads(response_text)
        
        # 2. 验证必填字段
        required_fields = ['brand_mentioned', 'rank', 'sentiment', 'cited_sources']
        for field in required_fields:
            if field not in geo_data:
                geo_data[field] = None
        
        # 3. 标准化字段
        geo_data['brand_mentioned'] = normalize_brand_mentioned(geo_data['brand_mentioned'])
        
        # 4. 返回结果
        return geo_data, None
        
    except Exception as e:
        api_logger.error(f"GEO 解析失败：{e}")
        return {
            'brand_mentioned': False,
            'rank': -1,
            'sentiment': 0.0,
            'cited_sources': [],
            'interception': '',
            '_error': str(e)
        }, str(e)


def calculate_result_quality(geo_data):
    """
    计算结果质量评分
    
    评分标准:
        - brand_mentioned: 10 分
        - rank > 0: 20 分
        - sentiment != 0: 20 分
        - cited_sources > 0: 30 分
        - interception 非空：20 分
    
    返回:
        {
            'quality_score': 0-100,
            'quality_level': 'high' | 'medium' | 'low' | 'failed',
            'quality_details': {...}
        }
    """
    if geo_data is None:
        return {'quality_score': 0, 'quality_level': 'failed', 'quality_details': {}}
    
    score = 0
    
    # brand_mentioned: 10 分
    if geo_data.get('brand_mentioned'):
        score += 10
    
    # rank > 0: 20 分
    if geo_data.get('rank', -1) > 0:
        score += 20
    
    # sentiment != 0: 20 分
    if geo_data.get('sentiment', 0) != 0:
        score += 20
    
    # cited_sources > 0: 30 分
    sources_count = len(geo_data.get('cited_sources', []))
    score += min(sources_count * 10, 30)
    
    # interception 非空：20 分
    if geo_data.get('interception'):
        score += 20
    
    # 确定质量等级
    if score >= 80:
        level = 'high'
    elif score >= 60:
        level = 'medium'
    elif score >= 40:
        level = 'low'
    else:
        level = 'failed'
    
    return {
        'quality_score': score,
        'quality_level': level,
        'quality_details': {
            'brand_mentioned': geo_data.get('brand_mentioned'),
            'rank': geo_data.get('rank'),
            'sentiment': geo_data.get('sentiment'),
            'sources': sources_count,
            'interception': geo_data.get('interception')
        }
    }
```

```python
WAL_DIR = '/tmp/nxm_wal'

def write_wal(execution_id, results, completed, total, brand=None, model=None):
    """
    预写日志 - 在内存持久化前写入磁盘
    
    目的：防止服务重启导致数据丢失
    """
    try:
        wal_path = os.path.join(WAL_DIR, f'nxm_wal_{execution_id}.pkl')
        wal_data = {
            'execution_id': execution_id,
            'results': results,
            'completed': completed,
            'total': total,
            'brand': brand,
            'model': model,
            'timestamp': time.time(),
            'last_updated': datetime.now().isoformat()
        }
        
        with open(wal_path, 'wb') as f:
            pickle.dump(wal_data, f)
        
        api_logger.info(f"[WAL] ✅ 已写入：{wal_path} (完成：{completed}/{total})")
        
    except Exception as e:
        api_logger.error(f"[WAL] ⚠️ 写入失败：{e}")
```

```python
class DiagnosisReportRepository:
    """诊断报告仓库 - 数据访问层"""
    
    @contextmanager
    def get_connection(self):
        """获取数据库连接（连接池）"""
        conn = get_db_pool().get_connection()
        try:
            yield conn
            conn.commit()
        except Exception as e:
            conn.rollback()
            raise
        finally:
            get_db_pool().return_connection(conn)
    
    def create(self, execution_id, user_id, config):
        """
        创建诊断报告
        
        参数:
            execution_id: 执行 ID
            user_id: 用户 ID
            config: 诊断配置 {
                brand_name: str,
                competitor_brands: list,
                selected_models: list,
                custom_questions: list
            }
        
        返回:
            report_id: int
        """
        # 检查是否已存在
        existing = self.get_by_execution_id(execution_id)
        if existing:
            return existing['id']
        
        now = datetime.now().isoformat()
        checksum = calculate_checksum({

        for i in range(0, len(results), batch_size):

                    ''', (...))
                    
                    result_ids.append(cursor.lastrowid)
                
                if commit:
```

```python
def _calculate_sentiment_distribution(self, results):
    """
    计算情感分布统计
    
    参数:
        results: 诊断结果列表
    
    返回:
        {
            'data': {
                'positive': 10,
                'neutral': 8,
                'negative': 7
            },
            'totalCount': 25,
            'averageSentiment': 0.35
        }

    sentiment_count = {'positive': 0, 'neutral': 0, 'negative': 0}
    sentiment_sum = 0
    
    for result in results:
        geo_data = result.get('geo_data', {})
        sentiment = geo_data.get('sentiment', 0)
        sentiment_sum += sentiment
        
        if sentiment > 0.3:
            sentiment_count['positive'] += 1
        elif sentiment < -0.3:
            sentiment_count['negative'] += 1
        else:
            sentiment_count['neutral'] += 1
    
    total = sum(sentiment_count.values())
    
    return {
        'data': sentiment_count,
        'totalCount': total,
        'averageSentiment': sentiment_sum / len(results) if results else 0
    }
```

```python
def _extract_keywords(self, results):
    """
    提取关键词
    
    参数:
        results: 诊断结果列表
    
    返回:
        [
            {'word': '品质', 'count': 15, 'sentiment': 0.5},
            {'word': '服务', 'count': 12, 'sentiment': 0.3},
            ...
        ]
    """
    keyword_stats = {}
    
    for result in results:
        geo_data = result.get('geo_data', {})
        keywords = geo_data.get('keywords', [])
        sentiment = geo_data.get('sentiment', 0)
        
        for kw in keywords:
            if kw not in keyword_stats:
                keyword_stats[kw] = {'count': 0, 'sentiment_sum': 0}
            
            keyword_stats[kw]['count'] += 1
            keyword_stats[kw]['sentiment_sum'] += sentiment
    
    # 转换为列表并排序
    keywords = [
        {
            'word': word,
            'count': stats['count'],
            'sentiment': stats['sentiment_sum'] / stats['count']
        }
        for word, stats in keyword_stats.items()
    ]
    
    # 按出现次数排序，取前 20 个
    keywords.sort(key=lambda x: x['count'], reverse=True)
    return keywords[:20]
```

```python
def _generate_brand_analysis(self, results, main_brand, competitor_brands):
    """
    生成品牌分析报告
    
    参数:
        results: 诊断结果列表
        main_brand: 主品牌
        competitor_brands: 竞品列表
    
    返回:
        {
            'brandAnalysis': {...},
            'userBrandAnalysis': {...},
            'competitorAnalysis': [...],
            'comparison': {...},
            'top3Brands': [...]
        }
    """
    # 1. 主品牌分析
    main_brand_results = [r for r in results if r.get('brand') == main_brand]
    user_brand_analysis = self._analyze_single_brand(main_brand_results, main_brand)
    
    # 2. 竞品分析
    competitor_analysis = []
    for competitor in competitor_brands:
        competitor_results = [r for r in results if r.get('brand') == competitor]
        analysis = self._analyze_single_brand(competitor_results, competitor)
        competitor_analysis.append(analysis)
    
    # 3. 对比分析
    comparison = self._generate_comparison(user_brand_analysis, competitor_analysis)
    
    # 4. Top3 品牌排名
    all_brands = [main_brand] + competitor_brands
    brand_scores = {}
    for brand in all_brands:
        brand_results = [r for r in results if r.get('brand') == brand]
        brand_scores[brand] = self._calculate_brand_score(brand_results)
    
    top3_brands = sorted(
        [{'brand': b, 'score': s} for b, s in brand_scores.items()],
        key=lambda x: x['score'],
        reverse=True
    )[:3]
    
    return {
        'brandAnalysis': {'main_brand': main_brand, 'analysis': user_brand_analysis},
        'userBrandAnalysis': user_brand_analysis,
        'competitorAnalysis': competitor_analysis,
        'comparison': comparison,
        'top3Brands': top3_brands
    }
```

```javascript
class PollingManager {
  /**
   * 开始轮询
   */
  startPolling(executionId, callbacks) {
    // 记录开始时间
    this.startTimes.set(executionId, Date.now());
    
    // 立即执行第一次轮询
    this._poll(executionId);
    
    return executionId;
  }
  
  /**
   * 内部轮询方法
   */
  _poll(executionId) {
    const task = this.pollingTasks.get(executionId);
    
    if (!task || !task.isActive) {
      return;
    }
    
    // 调用状态 API
    this._fetchStatus(executionId)
      .then(status => {
        // 回调状态更新
        task.callbacks.onStatus(status);
        
        // 检查是否完成
        if (status.status === 'completed' || status.status === 'failed') {
          task.callbacks.onComplete(status);
          this.stopPolling(executionId);
        } else {
          // 计算下次轮询间隔（指数退避）
          const interval = this._calculateInterval(status.progress, status.stage);
          task.timerId = setTimeout(() => this._poll(executionId), interval);
        }
      })
      .catch(error => {
        // 错误处理
        task.callbacks.onError(error);
        
        // 重试逻辑
        const retryCount = this.retryCounts.get(executionId) || 0;
        if (retryCount < this.maxRetries) {
          this.retryCounts.set(executionId, retryCount + 1);
          const backoffInterval = Math.min(
            this.baseInterval * Math.pow(2, retryCount),
            this.maxInterval
          );
          task.timerId = setTimeout(() => this._poll(executionId), backoffInterval);
        } else {
          task.callbacks.onTimeout();
          this.stopPolling(executionId);
        }
      });
  }
  
  /**
   * 获取状态
   */
  async _fetchStatus(executionId) {
    const res = await wx.request({
      url: `${app.globalData.serverUrl}/test/status/${executionId}`,
      method: 'GET'
    });
    
    return res.data;
  }
  
  /**
   * 计算轮询间隔
   */
  _calculateInterval(progress, stage) {
    // 根据阶段和进度动态调整间隔
    if (stage === 'init') {
      return this.stageIntervals.init;
    } else if (progress < 30) {
      return this.stageIntervals.fast;
    } else if (progress < 60) {
      return this.stageIntervals.medium;
    } else if (progress < 80) {
      return this.stageIntervals.slow;
    } else {
      return this.stageIntervals.final;
    }
  }
}
```

## 9. 常见问题排查

### 9.1 诊断无法完成

**现象**: 进度一直停留在某个百分比

**排查步骤**:
1. 检查后端日志：`backend_python/logs/app.log`
2. 查看 WAL 文件：`/tmp/nxm_wal/nxm_wal_{execution_id}.pkl`
3. 检查 AI 模型配额是否耗尽
4. 验证数据库连接池状态

**常见原因**:
- AI API 配额不足
- 网络连接超时
- 数据库锁死

### 9.2 报告数据为空

**现象**: `brandDistribution.data` 为空

**排查步骤**:
1. 查询数据库：`SELECT * FROM diagnosis_results WHERE execution_id = ?`
2. 检查 `geo_data` 解析是否成功
3. 验证 `extracted_brand` 字段是否有值

**修复方案**:
```sql
-- 从原始结果重建 brandDistribution
SELECT brand, COUNT(*) as count 
FROM diagnosis_results 
WHERE execution_id = 'xxx'
GROUP BY brand;
```

### 9.3 轮询无法停止

**现象**: 页面已关闭但轮询继续

**排查步骤**:
1. 检查 `onUnload` 是否调用 `stopPolling()`
2. 验证定时器是否被清除
3. 检查 `PollingManager.pollingTasks` 是否清理

**修复代码**:
```javascript
onUnload() {
  console.log('[DiagnosisPage] onUnload, cleaning up...');
  this.stopPolling();
  this.stopElapsedTimer();
}
```

### 9.4 数据库连接泄漏

**现象**: 请求变慢，数据库连接数持续增长

**排查步骤**:
1. 检查连接池状态：`get_db_pool().get_stats()`
2. 查看未归还的连接
3. 验证上下文管理器是否正确 `yield`

**修复方案**:
```python
@contextmanager
def get_connection(self):
    conn = get_db_pool().get_connection()
    try:
        yield conn
        conn.commit()
    except Exception as e:
        conn.rollback()
        raise
    finally:
        get_db_pool().return_connection(conn)  # 确保归还连接
```
