```javascript
/**
 * 【修复 5 - 2026-03-14】统一处理关键词格式
 */
_normalizeKeywords(keywords) {
  if (!keywords || !Array.isArray(keywords)) return [];
  
  if (typeof keywords[0] === 'string') {
    // 字符串数组转为对象数组
    return keywords.map(word => ({ word, weight: 1.0 }));
  } else if (typeof keywords[0] === 'object') {
    // 已经是对象数组
    return keywords;
  }
  
  return [];
}
```

**状态**: ✅ 已完成

**修复内容**:

- 指数退避（1s, 2s, 3s）

**文件**:
- `miniprogram/services/reportService.js` (Line 170-190)

**修复代码**:
```javascript
async getFullReport(executionId, options = {}) {
  const retryCount = options.retryCount || 0;
  const maxRetryCount = 3;
  const retryDelay = 1000 * (retryCount + 1); // 指数退避
  
  try {
    // ... API 调用
  } catch (error) {
    const errorType = this._classifyError(error);
    const canRetry = retryCount < maxRetryCount && 
                     errorType !== ErrorTypes.DATA_NOT_FOUND;
    
    if (canRetry) {
      await new Promise(resolve => setTimeout(resolve, retryDelay));
      return this.getFullReport(executionId, { retryCount: retryCount + 1 });
    }
    throw error;
  }
}
```

```javascript
async refreshReport() {
  this.setData({ isRefreshing: true });
  
  try {
    // 清除缓存
    reportService.clearCache(this.data.executionId);
    
    // 重新加载
    await this._loadFromAPI(this.data.executionId);
    
    wx.showToast({ title: '刷新成功', icon: 'success' });
  } catch (error) {
    wx.showToast({ title: '刷新失败', icon: 'none' });
  } finally {
    this.setData({ isRefreshing: false });
  }
}
```
