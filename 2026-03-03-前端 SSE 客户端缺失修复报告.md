```javascript
/**
 * SSE (Server-Sent Events) 客户端
 * 
 * 用于监听后端智能流水线更新事件
 * 
 * 功能:
 * 1. 建立 SSE 连接
 * 2. 监听事件
 * 3. 自动重连
 * 4. 错误处理
 */

const logger = require('./logger');
const config = require('./config');

/**
 * SSE 客户端类
 */
class SSEClient {
  constructor(url, options = {}) {
    this.url = url;
    this.options = options;
    this.eventSource = null;
    this.reconnectTimer = null;
    this.reconnectAttempts = 0;
    this.maxReconnectAttempts = options.maxReconnectAttempts || 5;
    this.reconnectDelay = options.reconnectDelay || 3000;
    this.onOpen = options.onOpen || (() => {});
    this.onMessage = options.onMessage || (() => {});
    this.onError = options.onError || (() => {});
    this.onClose = options.onClose || (() => {});
  }

  /**
   * 连接 SSE
   */
  connect() {
    if (this.eventSource) {
      logger.warn('[SSEClient] 已存在连接');
      return;
    }

    try {
      logger.info('[SSEClient] 开始连接:', this.url);
      
      // 微信小程序 WebSocket 连接
      this.eventSource = wx.connectSocket({
        url: this.url,
        success: () => {
          logger.info('[SSEClient] 连接请求已发送');
        },
        fail: (error) => {
          logger.error('[SSEClient] 连接失败:', error);
          this._handleError(error);
        }
      });

      // 监听 WebSocket 打开
      this.eventSource.onOpen(() => {
        logger.info('[SSEClient] 连接已建立');
        this.reconnectAttempts = 0;
        this.onOpen();
      });

      // 监听 WebSocket 消息
      this.eventSource.onMessage((res) => {
        try {
          const data = JSON.parse(res.data);
          logger.debug('[SSEClient] 收到消息:', data);
          this.onMessage(data);
        } catch (error) {
          logger.error('[SSEClient] 消息解析失败:', error);
        }
      });

      // 监听 WebSocket 错误
      this.eventSource.onError((error) => {
        logger.error('[SSEClient] 发生错误:', error);
        this._handleError(error);
      });

      // 监听 WebSocket 关闭
      this.eventSource.onClose(() => {
        logger.info('[SSEClient] 连接已关闭');
        this.eventSource = null;
        this.onClose();
        this._attemptReconnect();
      });

    } catch (error) {
      logger.error('[SSEClient] 连接异常:', error);
      this._handleError(error);
    }
  }

  /**
   * 尝试重新连接
   */
  _attemptReconnect() {
    if (this.reconnectAttempts >= this.maxReconnectAttempts) {
      logger.error('[SSEClient] 达到最大重连次数');
      this.onError(new Error('达到最大重连次数'));
      return;
    }

    this.reconnectAttempts++;
    const delay = this.reconnectDelay * this.reconnectAttempts;
    
    logger.info(`[SSEClient] 将在 ${delay}ms 后尝试重连 (${this.reconnectAttempts}/${this.maxReconnectAttempts})`);
    
    this.reconnectTimer = setTimeout(() => {
      this.connect();
    }, delay);
  }

  /**
   * 处理错误
   */
  _handleError(error) {
    this.onError(error);
  }

  /**
   * 关闭连接
   */
  close() {
    if (this.reconnectTimer) {
      clearTimeout(this.reconnectTimer);
      this.reconnectTimer = null;
    }

    if (this.eventSource) {
      this.eventSource.close({
        code: 1000,
        reason: '用户主动关闭'
      });
      this.eventSource = null;
    }

    logger.info('[SSEClient] 连接已关闭');
  }
}

/**
 * 创建 SSE 客户端并监听智能流水线更新
 * 
 * @param {string} executionId - 执行 ID
 * @param {function} onMessage - 消息回调
 * @param {function} onOpen - 连接打开回调
 * @param {function} onError - 错误回调
 * @returns {SSEClient} SSE 客户端实例
 */
function watchIntelligenceUpdates(executionId, onMessage, onOpen, onError) {
  if (!executionId) {
    throw new Error('executionId 不能为空');
  }

  // 构建 WebSocket URL
  const baseUrl = config.getApiBaseUrl();
  const wsUrl = baseUrl.replace('http://', 'ws://').replace('https://', 'wss://');
  const url = `${wsUrl}/ws/intelligence/${executionId}`;

  logger.info('[SSEClient] 创建智能监听:', url);

  const client = new SSEClient(url, {
    maxReconnectAttempts: 3,
    reconnectDelay: 2000,
    onOpen: () => {
      logger.info('[SSEClient] 智能监听已连接');
      if (onOpen) onOpen();
    },
    onMessage: (data) => {
      logger.debug('[SSEClient] 收到智能更新:', data);
      if (onMessage) onMessage({ type: 'intelligence', data });
    },
    onError: (error) => {
      logger.error('[SSEClient] 智能监听错误:', error);
      if (onError) onError(error);
    }
  });

  // 启动连接
  client.connect();

  return client;
}

module.exports = {
  SSEClient,
  watchIntelligenceUpdates
};
```

## 🔧 技术实现细节

### 1. WebSocket vs SSE

**选择 WebSocket 的原因**:
- 微信小程序环境对标准 SSE 支持有限
- WebSocket 提供双向通信能力
- 更好的错误处理和重连机制

### 2. URL 构建逻辑

```javascript
const baseUrl = config.getApiBaseUrl();  // 例如：http://localhost:5000
const wsUrl = baseUrl.replace('http://', 'ws://').replace('https://', 'wss://');
const url = `${wsUrl}/ws/intelligence/${executionId}`;
// 结果：ws://localhost:5000/ws/intelligence/{executionId}
```

### 3. 自动重连机制

| 参数 | 值 | 说明 |
|------|-----|------|
| `maxReconnectAttempts` | 3 | 最大重连次数 |
| `reconnectDelay` | 2000ms | 基础重连延迟 |
| 重连策略 | 递增延迟 | 每次重连延迟递增 |

### 4. 事件回调

```javascript
watchIntelligenceUpdates(
  executionId,      // 执行 ID
  onMessage,        // 消息回调：收到数据时触发
  onOpen,           // 连接回调：连接建立时触发
  onError           // 错误回调：发生错误时触发
)
```



### 2. 配置要求

`utils/config.js` 需要导出 `getApiBaseUrl()`:
```javascript
module.exports = {

```

生产环境需要在微信小程序后台配置 WebSocket 域名：
- 登录微信小程序后台
- 开发 → 开发设置 → 服务器域名
- 添加 `wss://your-domain.com`
