```python
async def _execute_in_transaction(
    self,
    operation_func: Callable,
    operation_name: str
) -> Any:
    """
    在独立短事务中执行操作

    【P0 修复 - 2026-03-05】缩小事务范围，每个操作独立提交

    参数:
        operation_func: 操作函数
        operation_name: 操作名称（用于日志）

    返回:
        操作结果
    """
    from wechat_backend.services.diagnosis_transaction import DiagnosisTransaction

    result = None
    try:
        # 创建独立事务
        tx = DiagnosisTransaction(
            execution_id=self.execution_id,
            execution_store=self.execution_store,
            auto_rollback=True
        )

        with tx:
            # 执行操作
            result = operation_func(tx)

            # 记录事务摘要
            summary = tx.get_summary()
            db_logger.info(
                f"[Orchestrator] 短事务完成：{operation_name}, "
                f"operations={summary['operation_count']}, "
                f"status={summary['status']}"
            )

    except Exception as e:
        db_logger.error(
            f"[Orchestrator] 短事务失败：{operation_name}, "
            f"error={e}"
        )
        raise

    return result
```

```python
def add_results_batch(
    self,
    report_id: int,
    results: List[Dict[str, Any]],
    batch_size: int = 10  # 【P0 新增】分批大小
) -> List[int]:
    """
    批量添加诊断结果（事务操作）

    【P0 修复 - 2026-03-05】支持分批提交，减少连接持有时间
    """
    self._init_dependencies()

    # 【P0 修复】分批执行
    all_result_ids = []
    for i in range(0, len(results), batch_size):
        batch = results[i:i + batch_size]
        is_last_batch = (i + batch_size >= len(results))

        # 执行批量添加（每批独立事务）
        result_ids = self._report_service.add_results_batch(
            report_id,
            self.execution_id,
            batch,
            commit=is_last_batch  # 最后一批才提交
        )
        all_result_ids.extend(result_ids)

        db_logger.info(
            f"[Transaction] 批量添加结果：batch={i//batch_size + 1}, "
            f"count={len(batch)}, total={len(all_result_ids)}"
        )

    # 定义回滚函数
    def rollback_add_results_batch():
        for rid in all_result_ids:
            try:
                self._result_repo.delete(rid)
            except Exception as e:
                db_logger.error(f"回滚失败：result_id={rid}, error={e}")

    # 记录操作
    self._log_operation(
        op_type=OperationType.ADD_RESULTS_BATCH,
        data={
            'report_id': report_id,
            'result_ids': all_result_ids,
            'count': len(results),
            'batches': (len(results) + batch_size - 1) // batch_size
        },
        rollback_func=rollback_add_results_batch,
        description=f"批量添加结果：count={len(results)}, batches={(len(results) + batch_size - 1) // batch_size}"
    )

    db_logger.info(
        f"[Transaction] 操作完成：批量添加结果 count={len(results)}, "
        f"batches={(len(results) + batch_size - 1) // batch_size}"
    )
    return all_result_ids
```

```python
def return_connection(self, conn):
    if conn is None:
        return

    conn_id = id(conn)

    with self._lock:
        if conn_id in self._in_use:
            self._in_use.discard(conn_id)
            self._pool.append(conn)

            # 【P1 新增】清理使用记录
            use_time = self._connection_use_time.pop(conn_id, None)
```

```python
def _check_connection_leak(self) -> List[int]:
    """
    检测连接泄漏

    返回:
        泄漏连接 ID 列表
    """
    current_time = time.time()
    leaked = []

    with self._lock:
        for conn_id, start_time in list(self._connection_use_time.items()):
            age = current_time - start_time
            if age > self.max_connection_age:
                thread_id = self._connection_thread.get(conn_id, 'unknown')
                db_logger.warning(
                    f"[连接泄漏] 连接超时未归还：id={conn_id}, "
                    f"年龄={age:.1f}秒，thread={thread_id}"
                )
                leaked.append(conn_id)

    return leaked

def _force_return_connection(self, conn_id: int):
    """
    强制归还连接

    参数:
        conn_id: 连接 ID
    """
    with self._lock:
        if conn_id in self._in_use:
            self._in_use.discard(conn_id)
            self._connection_use_time.pop(conn_id, None)
            self._connection_thread.pop(conn_id, None)
            self._leak_detection_count += 1

            db_logger.warning(
                f"[连接泄漏] 强制归还连接：id={conn_id}"
            )

def check_and_fix_leaks(self) -> int:
    """
    检查并修复连接泄漏

    返回:
        修复的泄漏连接数
    """
    leaked = self._check_connection_leak()
    for conn_id in leaked:
        self._force_return_connection(conn_id)
    return len(leaked)
```

```python
def get_metrics(self) -> Dict[str, Any]:
    """
    获取连接池指标

    返回:
        指标字典
    """
    with self._lock:
        utilization = len(self._in_use) / self.max_connections if self.max_connections > 0 else 0

        return {
            'active_connections': len(self._in_use),
            'available_connections': len(self._pool),
            'total_created': self._created_count,
```

#### 修改 B: 创建监控模块

**文件**: `monitoring/connection_pool_monitor.py` (新增)

```python
"""
连接池监控模块

功能:
1. 定期采集连接池指标
2. 发送告警（利用率>80% 警告，>90% 严重）
3. 记录监控历史
"""

import time
import threading
from datetime import datetime
from typing import Dict, Any, Optional
from wechat_backend.logging_config import api_logger


class ConnectionPoolMonitor:
    """连接池监控器"""

    def __init__(self, pool_getter):
        """
        初始化监控器

        参数:
            pool_getter: 获取连接池实例的函数
        """
        self._pool_getter = pool_getter
        self._history = []
        self._max_history = 1000
        self._stop_event = threading.Event()
        self._thread = None

        api_logger.info("[ConnectionPoolMonitor] 初始化完成")

    def start(self, interval_seconds: int = 10):
        """
        启动监控

        参数:
            interval_seconds: 采集间隔
        """
        self._thread = threading.Thread(
            target=self._monitor_loop,
            daemon=True,
            name="ConnectionPoolMonitor"
        )
        self._thread.start()
        api_logger.info(f"[ConnectionPoolMonitor] 监控已启动，间隔={interval_seconds}秒")

    def stop(self):
        """停止监控"""
        self._stop_event.set()
        if self._thread:
            self._thread.join(timeout=5.0)
        api_logger.info("[ConnectionPoolMonitor] 监控已停止")

    def _monitor_loop(self):
        """监控循环"""
        while not self._stop_event.is_set():
            try:
                pool = self._pool_getter()
                if pool:
                    metrics = pool.get_metrics()
                    self._record_metrics(metrics)
                    self._check_alerts(metrics)
            except Exception as e:
                api_logger.error(f"[ConnectionPoolMonitor] 采集失败：{e}")

            self._stop_event.wait(10.0)

    def _record_metric(self, metrics: Dict[str, Any]):
        """记录指标"""
        metrics['timestamp'] = datetime.now().isoformat()
        self._history.append(metrics)

        # 限制历史记录大小
        if len(self._history) > self._max_history:
            self._history = self._history[-self._max_history:]

    def _check_alerts(self, metrics: Dict[str, Any]):
        """检查告警"""
        if metrics.get('critical'):
            api_logger.error(
                f"[连接池告警] 严重：利用率={metrics['utilization']*100:.1f}%, "
                f"active={metrics['active_connections']}, "
                f"available={metrics['available_connections']}"
            )
        elif metrics.get('warning'):
            api_logger.warning(
                f"[连接池告警] 警告：利用率={metrics['utilization']*100:.1f}%, "
                f"active={metrics['active_connections']}"
            )

    def get_history(self, limit: int = 100) -> list:
        """获取历史记录"""
        return self._history[-limit:]


# 全局监控器实例
_monitor: Optional[ConnectionPoolMonitor] = None


def start_connection_pool_monitor(pool_getter, interval_seconds=10):
    """启动连接池监控"""
    global _monitor
    if _monitor is None:
        _monitor = ConnectionPoolMonitor(pool_getter)
        _monitor.start(interval_seconds)
    return _monitor


def get_connection_pool_metrics() -> Optional[Dict[str, Any]]:
    """获取最新指标"""
    global _monitor
    if _monitor and _monitor._history:
        return _monitor._history[-1]
    return None
```
