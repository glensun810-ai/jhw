### 方案 2：后端添加 AI 调用超时保护

async def execute(
    self,
    main_brand: str,
    competitor_brands: List[str],
    selected_models: List[Dict[str, Any]],
    raw_questions: List[str],
    user_id: str,
    user_level: str
) -> Dict[str, Any]:
    """
    执行并行诊断（带超时保护）
    """
    start_time = time.time()

    # 1. 初始化任务
    tasks = []
    self._total_tasks = len(raw_questions) * len(selected_models)

    api_logger.info(
        f"[NxM-Parallel] 🚀 开始并行执行 - execution_id={self.execution_id}, "
        f"问题数={len(raw_questions)}, 模型数={len(selected_models)}, "
        f"总任务数={self._total_tasks}, 最大并发={self.max_concurrent}"
    )

    # 2. 立即推送启动通知
    await self._push_progress(0, "ai_fetching", "正在启动 AI 调用")

    # 3. 创建所有任务（不等待）
    for q_idx, question in enumerate(raw_questions):
        for model_info in selected_models:
            model_name = model_info.get('name', '')

            # 创建异步任务（带超时保护）
            task = self._execute_single_task_with_timeout(  # ← 修改：使用带超时的方法
                question=question,
                model_name=model_name,
                q_idx=q_idx,
                model_idx=selected_models.index(model_info)
            )
            tasks.append(task)

    # 4. 并发执行所有任务（使用 gather，带总超时）
    api_logger.info(f"[NxM-Parallel] ⚡ 并发执行 {len(tasks)} 个任务...")

    try:
        # 【P0 关键修复】添加总超时保护（每个任务最多 60 秒）
        total_timeout = 60 * len(raw_questions) * len(selected_models) / self.max_concurrent
        api_logger.info(f"[NxM-Parallel] ⏰ 设置总超时：{total_timeout:.1f}秒")
        
        results = await asyncio.wait_for(
            asyncio.gather(*tasks, return_exceptions=True),
            timeout=total_timeout
        )
    except asyncio.TimeoutError:
        api_logger.error(f"[NxM-Parallel] ❌ 执行超时：{total_timeout}秒")
        # 返回已收集的结果，不等待所有任务完成
        return self._build_partial_result(
            f"执行超时（已收集 {len([r for r in results if isinstance(r, dict) and r.get('success')])}/{self._total_tasks} 个结果）"
        )
    except Exception as e:
        api_logger.error(f"[NxM-Parallel] ❌ 执行失败：{e}")
        return self._build_error_result(f"执行失败：{str(e)}")

    # 5. 处理结果（包括部分成功）

**新增方法**:

```python
async def _execute_single_task_with_timeout(
    self,
    question: str,
    model_name: str,
    q_idx: int,
    model_idx: int
) -> Dict[str, Any]:
    """
    执行单个任务（带超时保护）
    
    参数：
        question: 问题
        model_name: 模型名称
        q_idx: 问题索引
        model_idx: 模型索引
    
    返回：
        任务结果
    """
    try:
        # 获取超时配置
        timeout_manager = get_timeout_manager()
        timeout = timeout_manager.get_timeout(model_name) or 60  # 默认 60 秒
        
        # 使用 asyncio.wait_for 包装单个任务
        return await asyncio.wait_for(
            self._execute_single_task(question, model_name, q_idx, model_idx),
            timeout=timeout
        )
    except asyncio.TimeoutError:
        api_logger.error(
            f"[NxM-Parallel] ⏰ 任务超时：Q{q_idx + 1}×{model_name} "
            f"(>{timeout}秒)"
        )
        return {
            'success': False,
            'error': f'AI 调用超时（{timeout}秒）',
            'question': question,
            'model': model_name,
            'timeout': True
        }
    except Exception as e:
        api_logger.error(
            f"[NxM-Parallel] ❌ 任务失败：Q{q_idx + 1}×{model_name}, 错误={e}"
        )
        return {
            'success': False,
            'error': str(e),
            'question': question,
            'model': model_name
        }
```

### 方案 3：结果保存容错处理

**文件**: `wechat_backend/diagnosis_report_repository.py`

**修改位置**: 第 430-450 行

**修复内容**:

```python
def add_batch(self, report_id: int, execution_id: str,
             results: List[Dict[str, Any]]) -> List[int]:
    """
    批量添加诊断结果（容错处理）
    
    Args:
        report_id: 报告 ID
        execution_id: 执行 ID
        results: 结果列表
    
    Returns:
        成功插入的结果 ID 列表
    """
    result_ids = []
    failed_count = 0
    
    for idx, result in enumerate(results):
        try:
            result_id = self.add(report_id, execution_id, result)
            result_ids.append(result_id)
        except Exception as e:
            # 【P0 关键修复】单个结果保存失败不影响其他结果
            api_logger.error(
                f"[DiagnosisRepository] ❌ 结果 {idx + 1}/{len(results)} 保存失败："
                f"brand={result.get('brand', 'N/A')}, "
                f"question={result.get('question', 'N/A')[:50]}, "
                f"错误={e}"
            )
            failed_count += 1
            
            # 记录失败的结果（用于后续重试）
            result_ids.append(None)  # 用 None 占位
    
    # 记录批量保存摘要
    success_count = len([r for r in result_ids if r is not None])
    api_logger.info(
        f"[DiagnosisRepository] ✅ 批量保存完成："
        f"execution_id={execution_id}, "
        f"成功={success_count}/{len(results)}, "
        f"失败={failed_count}"
    )
    
    return result_ids
```

### 方案 4：验证逻辑降级处理

**文件**: `wechat_backend/services/diagnosis_orchestrator.py`

**修改位置**: 第 600-650 行

**修复内容**:

```python
async def _phase_results_validating(self, results: List[Dict[str, Any]]) -> PhaseResult:
    """
    阶段 4: 结果验证（容错处理）
    
    参数:
        results: AI 调用结果列表
    
    返回:
        PhaseResult
    """
    api_logger.info(f"[Orchestrator] 阶段 4: 结果验证 - {self.execution_id}")
    self.current_phase = DiagnosisPhase.RESULTS_VALIDATING
    
    try:
        # 使用 ResultValidator 进行完整验证
        from wechat_backend.services.result_validator import get_result_validator, ValidationLevel
        
        validator = get_result_validator(validation_level=ValidationLevel.NORMAL)
        validation_result = validator.validate(
            execution_id=self.execution_id,
            expected_results=results,
            validate_quality=True,
            validate_completeness=True
        )
        
        # 【P0 关键修复】检查是否为部分失败
        if not validation_result.is_passed:
            actual_count = validation_result.actual_count
            expected_count = validation_result.expected_count
            
            # 如果有部分结果（至少 50%），视为部分成功，继续流程
            if actual_count > 0 and actual_count >= expected_count * 0.5:
                api_logger.warning(
                    f"[Orchestrator] ⚠️ 结果不完整但继续：{self.execution_id}, "
                    f"成功={actual_count}/{expected_count} ({actual_count/expected_count:.1%})"
                )
                
                # 记录警告信息
                warning_msg = f"部分 AI 调用失败：{actual_count}/{expected_count}"
                api_logger.warning(f"[Orchestrator] {warning_msg}")
                
                return PhaseResult(
                    success=True,
                    data=validation_result.to_dict(),
                    warning=warning_msg  # 添加警告信息
                )
            
            # 结果太少（< 50%），视为失败
            error_msg = validation_result.message
            api_logger.error(f"[Orchestrator] ❌ 验证失败：{self.execution_id}, 错误={error_msg}")
            return PhaseResult(success=False, error=error_msg)
        
        # 验证通过
        api_logger.info(f"[Orchestrator] ✅ 验证通过：{self.execution_id}")
        return PhaseResult(
            success=True,
            data=validation_result.to_dict()
        )
        
    except Exception as e:
        api_logger.error(f"[Orchestrator] ❌ 验证失败：{self.execution_id}, 错误={e}", exc_info=True)
        return PhaseResult(success=False, error=str(e))
```
