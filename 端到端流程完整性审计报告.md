```javascript
async _poll(executionId) {
    const statusData = await this._pollViaCloudFunction(executionId);
    
    // 判断是否需要停止轮询
    if (statusData.should_stop_polling === true) {
        this.stopPolling(executionId);
        
        if (statusData.status === 'completed') {
            callbacks.onComplete(statusData);
        }
        return;
    }
    
    callbacks.onStatus(statusData);
}
```
